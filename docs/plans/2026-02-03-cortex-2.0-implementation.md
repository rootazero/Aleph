// core/src/dispatcher/cortex/parser/detector.rs

use serde_json::Value;

/// Streaming JSON detector states
#[derive(Debug, Clone, PartialEq)]
enum DetectorState {
    Scanning,
    InObject { depth: usize, in_string: bool, escape_next: bool },
    InArray { depth: usize, in_string: bool, escape_next: bool },
}

/// Parse result fragment
#[derive(Debug, Clone)]
pub enum JsonFragment {
    Complete(Value),
    Partial { prefix: String },
}

/// Streaming JSON detector with state machine
#[derive(Debug)]
pub struct JsonStreamDetector {
    state: DetectorState,
    buffer: String,
    start_char: Option<char>,
}

impl JsonStreamDetector {
    pub fn new() -> Self {
        Self {
            state: DetectorState::Scanning,
            buffer: String::new(),
            start_char: None,
        }
    }

    /// Push a chunk of text and return any complete JSON fragments
    pub fn push(&mut self, chunk: &str) -> Vec<JsonFragment> {
        let mut fragments = Vec::new();

        for ch in chunk.chars() {
            match &mut self.state {
                DetectorState::Scanning => {
                    if ch == '{' {
                        self.state = DetectorState::InObject {
                            depth: 1,
                            in_string: false,
                            escape_next: false,
                        };
                        self.buffer.push(ch);
                        self.start_char = Some('{');
                    } else if ch == '[' {
                        self.state = DetectorState::InArray {
                            depth: 1,
                            in_string: false,
                            escape_next: false,
                        };
                        self.buffer.push(ch);
                        self.start_char = Some('[');
                    }
                }

                DetectorState::InObject { depth, in_string, escape_next } => {
                    self.buffer.push(ch);

                    if *escape_next {
                        *escape_next = false;
                        continue;
                    }

                    if ch == '\\' && *in_string {
                        *escape_next = true;
                        continue;
                    }

                    if ch == '"' {
                        *in_string = !*in_string;
                        continue;
                    }

                    if !*in_string {
                        if ch == '{' {
                            *depth += 1;
                        } else if ch == '}' {
                            *depth -= 1;
                            if *depth == 0 {
                                if let Ok(value) = serde_json::from_str(&self.buffer) {
                                    fragments.push(JsonFragment::Complete(value));
                                }
                                self.buffer.clear();
                                self.state = DetectorState::Scanning;
                                self.start_char = None;
                            }
                        }
                    }
                }

                DetectorState::InArray { depth, in_string, escape_next } => {
                    self.buffer.push(ch);

                    if *escape_next {
                        *escape_next = false;
                        continue;
                    }

                    if ch == '\\' && *in_string {
                        *escape_next = true;
                        continue;
                    }

                    if ch == '"' {
                        *in_string = !*in_string;
                        continue;
                    }

                    if !*in_string {
                        if ch == '[' {
                            *depth += 1;
                        } else if ch == ']' {
                            *depth -= 1;
                            if *depth == 0 {
                                if let Ok(value) = serde_json::from_str(&self.buffer) {
                                    fragments.push(JsonFragment::Complete(value));
                                }
                                self.buffer.clear();
                                self.state = DetectorState::Scanning;
                                self.start_char = None;
                            }
                        }
                    }
                }
            }
        }

        fragments
    }

    /// Check if there's pending incomplete JSON
    pub fn has_pending(&self) -> bool {
        !self.buffer.is_empty()
    }

    /// Get current buffer for partial preview
    pub fn current_buffer(&self) -> &str {
        &self.buffer
    }

    /// Consume detector and return final buffer (for repair attempts)
    pub fn into_buffer(self) -> String {
        self.buffer
    }
}

impl Default for JsonStreamDetector {
    fn default() -> Self {
        Self::new()
    }
}

// core/src/dispatcher/cortex/parser/repair.rs

/// Attempt to repair truncated JSON
pub fn try_repair(incomplete: &str) -> Option<String> {
    if incomplete.is_empty() {
        return None;
    }

    let mut repaired = incomplete.to_string();

    // Step 1: Remove trailing dangling comma
    let trimmed = repaired.trim_end();
    if trimmed.ends_with(',') {
        repaired = trimmed[..trimmed.len() - 1].to_string();
    }

    // Step 2: Count unclosed brackets and quotes
    let mut brace_count: i32 = 0;
    let mut bracket_count: i32 = 0;
    let mut in_string = false;
    let mut escape_next = false;

    for ch in repaired.chars() {
        if escape_next {
            escape_next = false;
            continue;
        }

        if ch == '\\' && in_string {
            escape_next = true;
            continue;
        }

        if ch == '"' {
            in_string = !in_string;
            continue;
        }

        if !in_string {
            match ch {
                '{' => brace_count += 1,
                '}' => brace_count -= 1,
                '[' => bracket_count += 1,
                ']' => bracket_count -= 1,
                _ => {}
            }
        }
    }

    // Step 3: Close unclosed string
    if in_string {
        repaired.push('"');
    }

    // Step 4: Close unclosed brackets (in reverse order - ] before })
    for _ in 0..bracket_count {
        repaired.push(']');
    }

    for _ in 0..brace_count {
        repaired.push('}');
    }

    // Step 5: Validate result
    if serde_json::from_str::<serde_json::Value>(&repaired).is_ok() {
        Some(repaired)
    } else {
        None
    }
}

impl JsonStreamDetector {
    // ... existing methods ...

    /// Finalize detection, attempting repair on incomplete JSON
    pub fn finalize(self) -> Result<Vec<JsonFragment>, String> {
        if self.buffer.is_empty() {
            return Ok(vec![]);
        }

        // Try to parse as-is first
        if let Ok(value) = serde_json::from_str(&self.buffer) {
            return Ok(vec![JsonFragment::Complete(value)]);
        }

        // Attempt repair
        match try_repair(&self.buffer) {
            Some(repaired) => {
                if let Ok(value) = serde_json::from_str(&repaired) {
                    Ok(vec![JsonFragment::Complete(value)])
                } else {
                    Err(format!("Repair produced invalid JSON: {}", repaired))
                }
            }
            None => Err(format!("Could not repair JSON: {}", self.buffer)),
        }
    }
}

/// Result of sanitization
#[derive(Debug, Clone)]

/// Security configuration
#[derive(Debug, Clone, Default)]
pub struct SecurityConfig {

/// Unified trait for sanitization rules
pub trait SanitizerRule: Send + Sync {
    /// Rule name (for logging and config)
    fn name(&self) -> &str;

    /// Rule priority (lower executes first)
    fn priority(&self) -> u32;

    /// Execute sanitization
    fn sanitize(&self, input: &str, ctx: &SanitizeContext) -> SanitizeResult;

    /// Whether enabled
    fn is_enabled(&self, config: &SecurityConfig) -> bool {
        config.enabled
    }
}

/// Result of pipeline execution
#[derive(Debug)]
pub struct PipelineResult {
    pub text: String,
    pub actions: Vec<(String, SanitizeAction)>,
    pub blocked: bool,
    pub block_reason: Option<String>,
}

impl PipelineResult {
    pub fn ok(text: String, actions: Vec<(String, SanitizeAction)>) -> Self {
        Self {
            text,
            actions,
            blocked: false,
            block_reason: None,
        }
    }

    pub fn blocked(result: SanitizeResult) -> Self {
        let reason = match result.action {
            SanitizeAction::Blocked(ref r) => r.clone(),
            _ => "Unknown".to_string(),
        };
        Self {
            text: String::new(),
            actions: vec![],
            blocked: true,
            block_reason: Some(reason),
        }
    }
}

/// Security pipeline that chains rules
pub struct SecurityPipeline {
    rules: Vec<Box<dyn SanitizerRule>>,
    config: Arc<RwLock<SecurityConfig>>,
}

impl SecurityPipeline {
    pub fn new(config: SecurityConfig) -> Self {
        Self {
            rules: Vec::new(),
            config: Arc::new(RwLock::new(config)),
        }
    }

    pub fn add_rule(&mut self, rule: Box<dyn SanitizerRule>) {
        self.rules.push(rule);
        self.rules.sort_by_key(|r| r.priority());
    }

    pub fn process(&self, input: &str, ctx: &SanitizeContext) -> PipelineResult {
        let config = self.config.read();
        let mut current = input.to_string();
        let mut actions = Vec::new();

        for rule in self.rules.iter().filter(|r| r.is_enabled(&config)) {
            let result = rule.sanitize(&current, ctx);
            if result.triggered {
                actions.push((rule.name().to_string(), result.action.clone()));
                current = result.output;

                if matches!(result.action, SanitizeAction::Blocked(_)) {
                    return PipelineResult::blocked(result);
                }
            }
        }

        PipelineResult::ok(current, actions)
    }

    pub fn update_config(&self, config: SecurityConfig) {
        *self.config.write() = config;
    }
}

        "tag_injection"

    fn priority(&self) -> u32 {
        10 // Run early
    }

    fn sanitize(&self, input: &str, _ctx: &SanitizeContext) -> SanitizeResult {
        let mut output = input.to_string();
        let mut count = 0;
        let mut escaped_tags = Vec::new();

        for (pattern, name) in &self.patterns {
            let matches: Vec<_> = pattern.find_iter(&output).collect();

            }

        if count > 0 {
            SanitizeResult::escaped(
                output,
                count,
                Some(format!("Escaped tags: {:?}", escaped_tags)),
            )
        } else {
            SanitizeResult::pass(input)
        }
    }

    fn is_enabled(&self, config: &SecurityConfig) -> bool {
        config.enabled && config.tag_injection_enabled
    }
}

/// Decision action based on confidence
#[derive(Debug, Clone, PartialEq)]
pub enum DecisionAction {
    NoMatch,
    RequiresConfirmation,
    OptionalConfirmation,
    AutoExecute,
}

impl DecisionConfig {
    /// Decide action based on confidence score
    pub fn decide(&self, confidence: f32) -> DecisionAction {
        if confidence < self.routing.no_match_threshold {
            DecisionAction::NoMatch
        } else if confidence < self.confirmation.require_threshold {
            DecisionAction::RequiresConfirmation
        } else if confidence < self.confirmation.auto_execute_threshold {

        std::fs::write(path, content)
    }

// core/src/dispatcher/cortex/decision/session_override.rs

use serde::{Deserialize, Serialize};
use super::config::DecisionConfig;

/// Session-level config override
#[derive(Debug, Clone, Default, Serialize, Deserialize)]
pub struct SessionOverride {
    pub routing: Option<RoutingOverride>,
    pub confirmation: Option<ConfirmationOverride>,
}

#[derive(Debug, Clone, Serialize, Deserialize)]
pub struct RoutingOverride {
    pub l3_enabled: Option<bool>,
    pub l3_timeout_ms: Option<u64>,
}

#[derive(Debug, Clone, Serialize, Deserialize)]
pub struct ConfirmationOverride {
    pub enabled: Option<bool>,
    pub auto_execute_threshold: Option<f32>,
}

/// Merge global config with session override
pub fn merge_config(global: &DecisionConfig, session: &SessionOverride) -> DecisionConfig {
    let mut merged = global.clone();

    if let Some(ref routing) = session.routing {
        if let Some(timeout) = routing.l3_timeout_ms {
            merged.routing.l3_timeout_ms = timeout;
        }
    }

    if let Some(ref confirmation) = session.confirmation {
        if let Some(enabled) = confirmation.enabled {
            merged.confirmation.enabled = enabled;
        }
        if let Some(threshold) = confirmation.auto_execute_threshold {
            merged.confirmation.auto_execute_threshold = threshold;
        }
    }

    merged
}
